    loadLogic: function () {
        let intervalid;
        const datoUsuario = JSON.parse(localStorage.getItem('usuario-progreso'));
        console.log(datoUsuario);
        const juego_control = {
            estrellas: 0,
            vida: 3,
            tiempo: 3,
        }
        const btnContinuar = document.querySelector('.continuar'); 
        btnContinuar.addEventListener('click', e => {
            new ScenesManager().cargarPantalla( mapaniveles );
        });

        
        const hoveraudio = new Audio('/build/assets/audio/pop.mp3');
        const incorrecto= new Audio('/build/assets/audio/incorrecto.mp3');
        const correcto= new Audio('/build/assets/audio/correcto.mp3');
        let estaHablando = false;

        const calcularResultado = () => {
            if(juego_control.vida == 3 && juego_control.tiempo <= 3) juego_control.estrellas = 3; 
            if(juego_control.vida <= 2 && juego_control.tiempo >= 2) juego_control.estrellas = 2; 
            if(juego_control.vida <= 1 && juego_control.tiempo >= 2) juego_control.estrellas = 2; 
            if(juego_control.vida == 1 && juego_control.tiempo >= 1) juego_control.estrellas = 1; 
            if(juego_control.vida == 1 && juego_control.tiempo >= 0) juego_control.estrellas = 1; 
        }


        const nivelFallido = () => {
            console.log('nivel fallido');
            const modal = document.querySelector('.modal');
            const pantallaResultado = document.querySelector('.resultado');
            const textoResultado = document.querySelector('.resultado-txt');
            modal.classList.remove('hide');
            pantallaResultado.classList.remove('hide');
            textoResultado.classList.add('fallido');
            textoResultado.innerHTML = 'Nivel Fallido';
            clearInterval(intervalid);
        }


        const nivelCompletado = () => {
            console.log(juego_control);
            const modal = document.querySelector('.modal');
            const pantallaResultado = document.querySelector('.resultado');
            const textoResultado = document.querySelector('.resultado-txt');
            const estrellas = document.querySelector('.estrellas-resultado');
            if(juego_control.estrellas == 1) estrellas.innerHTML = '⭐';
            if(juego_control.estrellas == 2) estrellas.innerHTML = '⭐⭐';
            if(juego_control.estrellas == 3) estrellas.innerHTML = '⭐⭐⭐';
            modal.classList.remove('hide');
            pantallaResultado.classList.remove('hide');
            textoResultado.classList.add('fallido');
            textoResultado.innerHTML = 'Nivel Completado';
            guardarResultado();
        }
        

        const restarVida = () => {
            juego_control.vida--;
           const cuadrovida =  document.querySelector('.vida');
           const vida =  document.querySelector('.vida-jugador');
           cuadrovida.classList.add('malo');
           vida.innerHTML = vida.innerHTML.slice(0, -2);
              setTimeout(() => {
                cuadrovida.classList.remove('malo');
              }, 500);
        }


        const hablar = (texto) => {
            const personaje = document.querySelector('.personaje');
            const hablarSonido = new Audio('/build/assets/audio/hablar.mp3');
            const textoHablar = document.querySelector('.texto');

            hablarSonido.load();
            textoHablar.innerHTML = texto;
            textoHablar.classList.add('hablar');
            personaje.classList.add('hablar-pj');
            hablarSonido.play();
            setTimeout(() => {
            hablarSonido.pause();
            personaje.classList.remove('hablar-pj');
            textoHablar.classList.remove('hablar');
            estaHablando = false;
            }, 4000);
        }

        const iniciarTiempo = (minutos) => {
            let minute = minutos;
            let sec = 30;
            intervalid = setInterval(function() {
                let random = Math.floor(Math.random() * 30) + 1;
                if(random == 7 && !estaHablando) {hablar('voy a reprobar el año, NOOOOOO'); estaHablando = true;}
                if(random == 14 && !estaHablando) {hablar('no se que hacer, no se que hacer, si no resuelvo esto <br> me van a hechar de la casa'); estaHablando = true;}
                if(juego_control.vida == 0) nivelFallido();
               
             document.querySelector(".tiempo-juego").innerHTML = minute + ":" + sec;
            sec--;
            if (sec == 0) {
                if (minute == 0 && sec ==0) nivelFallido();
            minute--;
            juego_control.tiempo = minute;
            sec = 60;

    
         }
            }, 1000);
          
        }

        const setearBotones = () => {
            const botones = document.querySelectorAll('.opc')

            const respuestaMala = (e) => {
                incorrecto.load();
                incorrecto.play();
                restarVida();
            }

            for(const boton of botones){
                boton.addEventListener('mouseover', e => {
                    hoveraudio.load();
                    hoveraudio.play();
                })

                boton.addEventListener('click', respuestaMala)
            }
            //respuesta correcta
            botones[0].removeEventListener('click',  respuestaMala)
            botones[0].addEventListener('click', e => {
                correcto.load();
                correcto.play();
                clearInterval(intervalid);
                calcularResultado();
                nivelCompletado();
            })
            console.log(botones);
        }

        const guardarResultado = () => {
                if(!datoUsuario.progreso[1]) datoUsuario.progreso.push({nivel:2,estrellas:0})
                if(datoUsuario.progreso[0].estrellas < juego_control.estrellas) datoUsuario.progreso[0].estrellas = juego_control.estrellas;
                localStorage.setItem('usuario-progreso', JSON.stringify(datoUsuario))
        }

        //iniciar juego
        setTimeout(() => {
            iniciarTiempo(3);
            const modal = document.querySelector('.modal');
            const modalTexto = document.querySelector('.preparado');
            modalTexto.classList.add('hide');
            modal.classList.add('hide');
            setearBotones();
            hablar('Hola, soy el personaje de este juego');
            setTimeout(() => {
               hablar('Hola necesito ayuda encontrando cuantas variables hay en <br> este problema si no los encuentro mi profesora <br>me va a matar!! ayudame porfavor!!');
            
            }, 5000);

        }, 5000)
      
         
    },

    onDestroy: function () {
        const modal = document.querySelector('.modal');
        const modalTexto = document.querySelector('.preparado');
        const pantallaResultado = document.querySelector('.resultado');
        pantallaResultado.classList.add('hide');
        modalTexto.classList.add('hide');
        modal.classList.add('hide');
        console.log('destruyendo juego 1');
    }
